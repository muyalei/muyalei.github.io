```
#from math import sqrt,sin,atan2,cos
    
    #中国正常GCJ02坐标---->百度地图BD09坐标
    #腾讯地图用的也是GCJ02坐标
    def trans_tx2bd(self,position):
        params = position.split(',')
        lng = params[0]
        lat = params[1]

        x_pi = 3.14159265358979324*3000.0/180.0
        x = lng
        y = lat
        z = sqrt(x*x+y*y)+0.00002*sin(y*x_pi)
        theta = atan2(y,x)+0.000003*cos(x*x_pi)
        lng2 = z*cos(theta)+0.0065
        lat2 = z*sin(theta)+0.006

        return str(lng2)+','+str(lat2)
```
